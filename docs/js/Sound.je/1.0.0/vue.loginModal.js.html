// @flow
// Node Modules
const Vue = require(`vue/dist/vue`);

// Site Modules
const Util = require(`${__base}/core/Util`);
const Ajax = require(`${__base}/core/Ajax`);
const store = require(`${__base}/vue/store`);
const { Modal } = require(`${__base}/core/Router/Enum`);
const mixinModal = require(`${__base}/vue/mixins/modal`);
const router = require(`${__base}/core/Router`);

const vueLoginModal = Vue.extend({
    mixins: [
        mixinModal
    ],
    /**
     * The components design template, see the "text/x-template" tags in
     * index.html
     */
    template: `#loginModal-template`,
    /**
     * The components variables that it inherits from its parent
     */
    props: [],
    /**
     * Every component requires a data feild, but i do not need it as
     * i inherit all the relevent data. So it is filled with nonsense.
     */
    data(): Object {
        return {
            errorTitle: null,
            errorMsg: null
        };
    },
    computed: {},
    methods: {
        closeModal() {
            store.commit(`setModal`, {
                modal: Modal.NONE
            });
        },
        onLogin(request: Promise<string>) {
            this.errorTitle = null;
            this.errorMsg = null;
            request
                .then((response: string) => {
                    store.dispatch(`processSession`, response);
                    router.redirect(`/`);
                })
                .catch((response: string) => {
                    const data = JSON.parse(response);
                    this.errorTitle = data.error;
                    this.errorMsg = data.message;
                });
        }
    }
});

module.exports = vueLoginModal;
