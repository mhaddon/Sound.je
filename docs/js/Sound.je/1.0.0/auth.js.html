// @flow
// Node modules

// Site Modules
const store = require(`${__base}/vue/store`);

const authMixin = {
    data() {
        return {
            /**
             * How often in ms will we poll the server for session information
             */
            pollFrequency: 30000
        };
    },
    created() {
        this.pollSession();
    },
    methods: {
        pollSession() {
            store.dispatch(`updateSession`)
                .then(() => {
                    window.setTimeout(() => {
                        /**
                         * requestAnimationFrame ensures we only poll when the window has focus
                         */
                        window.requestAnimationFrame(this.pollSession.bind(this));
                    }, this.pollFrequency);
                });
        }
    }
};

module.exports = authMixin;
