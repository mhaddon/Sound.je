// @flow
// Node Modules
const Vue = require(`vue/dist/vue`);

// Site Modules
const youtubeBox = require(`${__base}/vue/components/vue.youtubeBox`);
const soundcloudBox = require(`${__base}/vue/components/vue.soundcloudBox`);
const facebookBox = require(`${__base}/vue/components/vue.facebookBox`);
const PlaylistController = require(`${__base}/core/musicplayer/PlaylistController`);
const MusicPlayer = require(`${__base}/core/musicplayer`);

const vueMediaBox = Vue.extend({
    /**
     * The components design template, see the "text/x-template" tags in
     * index.html
     */
    template: `#mediaBox-template`,
    /**
     * The components variables that it inherits from its parent
     */
    props: [`Medium`, `group`],
    /**
     * Every component requires a data feild, but i do not need it as
     * i inherit all the relevent data. So it is filled with nonsense.
     */
    components: {
        "youtubebox-component": youtubeBox,
        "soundcloudbox-component": soundcloudBox,
        "facebookbox-component": facebookBox
    },
    data(): Object {
        return {
            loaded: false
        };
    },
    created() {
        this.$nextTick(() => {
            if (this.group) {
                PlaylistController.get(this.group).add(this);
            }
        });
    },
    beforeDestroy() {
        if (this.$el.getAttribute(`data-group`)) {
            PlaylistController.get(this.group).remove(this);
        }
    },
    computed: {
        type(): number | boolean {
            return (this.loaded === true) ? this.Medium.type : false;
        }
    },
    methods: {
        play() {
            MusicPlayer.unload();
            this.loaded = true;
            MusicPlayer.setPlaylist(this.group);
            MusicPlayer.unload = () => {
                this.loaded = false;
            };
        }
    }
});

module.exports = vueMediaBox;
